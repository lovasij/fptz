Adatbázis létrehozása

Páciens

CREATE TABLE 212e_orvos.paciens (
pid CHAR(11) PRIMARY KEY,
nev VARCHAR(50),
cim VARCHAR(100),
telefon VARCHAR(15)
);

Betegseg

CREATE TABLE 212e_orvos.betegseg (
bnev VARCHAR(50) PRIMARY KEY
);

Gyógyszer

CREATE TABLE 212e_orvos.gyogyszer (
gynev VARCHAR(50) PRIMARY KEY
);

Mit kap

CREATE TABLE 212e_orvos.mit_kap (
bnev VARCHAR(50),
gynev VARCHAR(50),
PRIMARY KEY (bnev, gynev),
FOREIGN KEY (gynev) REFERENCES 212e_orvos.gyogyszer (gynev),
FOREIGN KEY (bnev) REFERENCES 212e_orvos.betegseg (bnev)
);

Vizsgálat

CREATE TABLE 212e_orvos.vizsgalat (
vid INT AUTO_INCREMENT,
pid CHAR(11),
datum DATE,
PRIMARY KEY(vid, pid, datum),
FOREIGN KEY (pid) REFERENCES 212e_orvos.paciens (pid)
);

Mire érzékeny

CREATE TABLE 212e_orvos.mire_erzekeny (
pid CHAR(11),
gynev VARCHAR(50),
PRIMARY KEY(pid, gynev),
FOREIGN KEY (pid) REFERENCES 212e_orvos.paciens (pid),
FOREIGN KEY (gynev) REFERENCES 212e_orvos.gyogyszer (gynev)
);

Diagnózis

CREATE TABLE 212e_orvos.diagnozis (
id INT AUTO_INCREMENT PRIMARY KEY,
vid INT,
datum DATE,
bnev VARCHAR(50),
FOREIGN KEY (vid) REFERENCES 212e_orvos.vizsgalat (vid),
FOREIGN KEY (bnev) REFERENCES 212e_orvos.betegseg (bnev)
)
